6.删除当前目录下大小为0的文件

```shell
#/bin/bash  
for filename in `ls`  
do  
if test -d $filename  
then b=0

else   
a=$(ls -l $filename | awk '{ print $5 }')  
if test $a -eq 0  
then rm $filename  
fi  
fi   
done
```

28.检查端口号是否已启动  

```shell
#!/bin/bash  
n=1  
echo "检查xxx服务..."  
while true  
do  
if test $n -gt 20  
then   
echo "xxx服务启动失败"  
break  
fi  
  
sleep 5  
n=$(($n+1))  
port=`netstat -antp | grep "0.0.0.0:8080"`  
if [ ${#port} -gt 3 ]; then  
echo "xxx服务已经启动"  
break;  
fi  
done
```
